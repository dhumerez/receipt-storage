Add state: `const [pdfLoading, setPdfLoading] = useState(false);`

Add handler:
```typescript
async function handleDownloadReceipt() {
  if (!transaction) return;
  setPdfLoading(true);
  try {
    await downloadPdf(
      `/api/v1/reports/receipt/${transaction.id}/pdf`,
      `receipt-${transaction.referenceNumber}.pdf`
    );
  } catch {
    // Could show toast/alert — for now just reset loading
  } finally {
    setPdfLoading(false);
  }
}
```

async function handleDownloadStatement() {
  if (!debt) return;

      `/api/v1/reports/receipt/${debt.transactionId}/pdf`,

    setPdfLoading(false);
  }
}
