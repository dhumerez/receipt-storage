export function getTransactions(params: { search?, clientId?, status?, dateFrom?, dateTo? }): Promise<TransactionListItem[]>
export function getTransaction(id: string): Promise<TransactionDetail>
export function getFileUrl(filePath: string): string

export interface TransactionListItem {
  id, referenceNumber, clientName, totalAmount (string), initialPayment (string),
  status ('draft'|'pending_approval'|'active'|'voided'|'written_off'),
  deliveredAt (string|null), submittedBy, createdAt
}

export interface TransactionDetail extends TransactionListItem {
  clientId, description, clientNotes, internalNotes,
  items: TransactionLineItem[], documents: DocumentInfo[]
}

export interface DocumentInfo {
  id, filePath, originalName, mimeType, fileSizeBytes
}

- Section heading: "Payment Summary" `text-sm font-semibold text-gray-700 mb-3`.
- Initial payment: `$${initialPayment}`.

- No remove buttons on detail view (unlike the create form).
- If no documents: "No attachments" in `text-sm text-gray-500`.

**Section 5 --- Notes:** (`border-t border-gray-200 mt-8 pt-8`)
- Client Notes: if present, heading "Client Notes" + `text-sm text-gray-700` paragraph.

- If no notes at all: "No notes" in `text-sm text-gray-500`.

Loading state: skeleton or "Loading..." text.
